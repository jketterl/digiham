SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${PROJECT_SOURCE_DIR}/cmake/")

add_executable(rrc_filter rrc_filter.c)

add_executable(gfsk_demodulator gfsk_demodulator.c)
